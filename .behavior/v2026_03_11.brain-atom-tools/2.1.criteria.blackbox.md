# blackbox criteria: brain atom tools

## usecase.1 = tool invocation

given('a brain atom with tools plugged')
  when('asked a prompt that requires tool use')
    then('output is null')
      sothat('caller knows brain deferred to tools')
    then('calls.tools contains invocations')
      sothat('caller can execute the tools')
    then('each invocation has exid, slug, and typed input')
      sothat('caller can correlate results and validate input')

given('a brain atom with tools plugged')
  when('asked a prompt that does not require tool use')
    then('output is defined')
      sothat('caller receives direct answer')
    then('calls is null')
      sothat('caller knows no tools were requested')

## usecase.2 = tool result continuation

given('a brain atom with prior tool invocations')
  when('continued with BrainPlugToolExecution[] as prompt')
    then('brain receives tool results')
      sothat('brain can reason about the results')
    then('output is defined if brain is satisfied')
      sothat('episode can complete')
    then('calls.tools has new invocations if brain needs more')
      sothat('caller can continue the loop')

given('a tool execution with signal: success')
  when('continued via episode')
    then('brain uses the output value')
      sothat('successful results inform the response')

given('a tool execution with signal: error:constraint')
  when('continued via episode')
    then('brain reasons about the constraint violation')
      sothat('brain can adjust its approach')

given('a tool execution with signal: error:malfunction')
  when('continued via episode')
    then('brain reasons about the failure')
      sothat('brain can report or retry')

## usecase.3 = structured output with tools

given('a brain atom with tools and output schema')
  when('brain uses tools and produces final answer')
    then('output conforms to schema')
      sothat('caller receives typed response')

given('a brain atom with tools but no output schema')
  when('brain uses tools and produces final answer')
    then('output is unstructured text')
      sothat('schema-less workflows are supported')

## usecase.4 = parallel tool calls

given('a brain atom with multiple tools plugged')
  when('asked a prompt that needs multiple tools')
    then('calls.tools contains multiple invocations')
      sothat('caller can execute all in parallel')
    then('each invocation has unique exid')
      sothat('results can be correlated')

given('multiple tool executions')
  when('all passed as BrainPlugToolExecution[] prompt')
    then('brain receives all results')
      sothat('brain can synthesize across tools')

## usecase.5 = episode and metrics

given('any tool use workflow')
  when('completed')
    then('episode.exid is returned')
      sothat('caller can continue or audit')
    then('metrics include token usage and cost')
      sothat('caller can observe resource consumption')
